Exer2Alg

Ler um valor interio e indicar se é par ou impar

START
	valor: inteiro;

	Escrever "Digite um valor inteiro"
	Ler valor
	
	Se (valor MOD 2 = 0) então
		Escrever valor, "é par"
	Senão
		Escever valor, "é impar"
	FimSE
END
